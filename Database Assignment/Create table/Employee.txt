Create table Employee (
	employee_id   varchar(5)  not null,
	employee_Fname  varchar(10)  not null,
	employee_Lname  varchar(10),
	employee_salary   number(10,2)  not null,
	employee_phone  varchar(12) not null,
	employee_email  varchar(30),
	hotel_id  varchar(5)   not null,
Primary key (employee_id),
Foreign key (hotel_id)  references  Hotel (hotel_id),
Constraint chk_employee_phone check (employee_phone like '___-%'),
Constraint chk_employee_email check (employee_email like '%@%.com')
);
