Create table Room (
	Room_id  varchar(5)  not null,
	Room_type  varchar(30)  not null,
	Adult_capacity  number  not null,
	Child_capacity   number   not null,
	Room_price   number(10,2)  not null,
	Room_desc   varchar(80),
	Room_status   varchar(3)  not null,
	Hotel_id   varchar(5)  not null,
Primary key(room_id),
Foreign key(hotel_id) references Hotel(hotel_id),
Constraint chk_room_status check (UPPER(room_status) in ('YES','NO'))
);
