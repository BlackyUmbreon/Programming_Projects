Create table Customer (
	cust_id   varchar(5)    not null,
	cust_Fname  varchar(10)   not null,
	cust_Lname  varchar(10) ,
	cust_phone  varchar(12),
	cust_email  varchar(30) ,
	address_id  varchar(8)   not null,
Primary key (cust_id),
Foreign key (address_id) references Address (address_id),
Constraint chk_cust_phone check (cust_phone like '___-%'),
Constraint chk_cust_email check (cust_email like '%@%.com')
);
