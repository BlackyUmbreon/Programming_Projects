TTITLE Center 'Search Customer phone number with booking id entered'
Select C.cust_id ,C.cust_fname,C.cust_Lname, C.cust_Phone, B.booking_id, B.booking_status
From customer C, Booking B 
Where (C.cust_id = B.cust_id AND C.cust_fname = '&v_cust_fname') AND UPPER(B.booking_status) = UPPER('&v_booking_status');
