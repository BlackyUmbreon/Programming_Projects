cl scr
PROMPT 'Finding Booking details with rounding entered'
PROMPT
PROMPT
ACCEPT v_rounding  number FORMAT '99999.99'   PROMPT 'Enter Rounding                      :'
PROMPT
PROMPT

Set linesize 100
Column booking_status format A15
Column payment_id format A20
Column payment_type format A20
Column booking_id format A20
TTITLE Center 'Search Booking details with rounding lists'
Select P.Payment_ID, P.Payment_Type, P.Rounding, B.Booking_ID, B.Booking_Status 
From payment P, booking B 
Where P.Rounding<=&v_rounding AND B.Booking_ID=P.Booking_ID
order by P.rounding;
TTitle off;
clear column
