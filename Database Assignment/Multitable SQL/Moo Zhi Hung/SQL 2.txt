PROMPT 'Finding Room price'
PROMPT
PROMPT
ACCEPT v_start_room_price  number  FORMAT '9999.99' PROMPT 'Enter least Room Price                       :'
ACCEPT v_end_room_price  number  FORMAT '9999.99'   PROMPT 'Enter most Room Price                        :'
ACCEPT v_start_booking_date  date                   PROMPT 'Enter Starting Booking Date (DD-MMM-YYYY)    :'
ACCEPT v_end_booking_date  date                     PROMPT 'Enter Ending   Booking Date (DD-MMM-YYYY)    :'
PROMPT
PROMPT

set linesize 120
set pagesize 100
column room_type format A20
column room_status format A20
column room_price format '9999.99'
TTITLE Center 'Room Search For'
select R.room_id,R.room_type,R.adult_Capacity,R.child_Capacity,R.room_status,B.Booking_date,R.room_price
from room R, booking_detail B
where (R.room_id = B.room_id AND B.Booking_date between '&v_start_booking_date' AND '&v_end_booking_date') AND R.room_price BETWEEN  &v_start_room_price AND &v_end_room_price
order by R.room_type, B.booking_date , R.room_price;
TTitle off;
clear column;
