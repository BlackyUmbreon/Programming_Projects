PROMPT 'Finding Booking details from the tax rate'
PROMPT
PROMPT
ACCEPT v_tax_rate  number  FORMAT '99999.99'   PROMPT   'Enter tax rate                      :'
ACCEPT v_booking_status char  FORMAT 'A30'   PROMPT 'Enter Booking Status                           :'
PROMPT
PROMPT

Set linesize 140
Column booking_status format A15
Column payment_id format A15
Column payment_type format A20
Column booking_id format A20
column tax_rate format '999'
TTITLE Center 'Search Booking details from the tax rate'
Select P.Payment_ID,P.Tax_Rate*100 as Tax_Rate, B.Booking_Status , A.*
From payment P, booking B , booking_detail A
Where ((B.Booking_ID=P.Booking_ID AND UPPER(B.booking_status) = UPPER('&v_booking_status')) AND B.booking_ID = A.booking_id) AND P.tax_rate = &v_tax_rate
order by A.booking_date;
TTitle off;
clear column
