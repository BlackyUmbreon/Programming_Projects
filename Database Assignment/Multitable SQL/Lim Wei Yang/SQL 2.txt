cl scr
PROMPT 'Finding Customer'
PROMPT
PROMPT
ACCEPT v_cust_Fname               char    FORMAT 'A30'               PROMPT 'Enter Customer First Name                :'
ACCEPT v_start_booking_date       date                               PROMPT 'Enter Starting Booking Date (DD-MMM-YYYY):'
ACCEPT v_end_booking_date         date                               PROMPT 'Enter Ending   Booking Date (DD-MMM-YYYY):'
PROMPT
PROMPT

set linesize 140
column cust_phone format 'A15'
column cust_email format 'A40'
TTITLE Center 'Customer Search For'
select C.cust_id,C.cust_fname,C.cust_lname,C.cust_phone,C.cust_email,B.Booking_date
from customer C, booking_detail B, booking A
where ((C.cust_Fname = '&v_cust_Fname' AND C.cust_id = A.cust_id) AND A.booking_id = B.booking_id) AND B.Booking_date between '&v_start_booking_date' AND '&v_end_booking_date'
order by B.booking_date;
clear column;
TTitle off;
