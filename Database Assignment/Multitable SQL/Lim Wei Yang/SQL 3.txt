from room R, booking_detail B, booking A
where (((UPPER(R.room_status) = UPPER('&v_room_status') AND R.room_id = B.room_id) AND B.Booking_date between '&v_start_booking_date' AND '&v_end_booking_date') AND B.booking_ID = A.booking_ID) AND UPPER(A.booking_status) = UPPER('&v_booking_status')
