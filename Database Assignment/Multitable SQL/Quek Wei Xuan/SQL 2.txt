cl scr
PROMPT 'Search Customers with Payment Type'
PROMPT
PROMPT
ACCEPT v_Payment_Type  char  FORMAT 'A30'   PROMPT 'Enter Payment Type                        :'
PROMPT
PROMPT

Set linesize 180
Set pagesize 30 
Column Cust_ID format A10
Column Cust_FName format A20
Column Cust_LName format A20
Column Cust_Phone format A15
Column Cust_Email format A30
Column Booking_ID format A15
Column Payment_Type format A15
Column Payment_Amount format 999999.99
TTITLE Center 'Search Customer Detail with specific Payment Type and Booking Date Range'
Select C.Cust_ID, C.Cust_FName, C.Cust_LName, Cust_Phone, Cust_Email, B.Booking_ID, P.Payment_Type, P.Payment_Amount  
From payment P, booking B, customer C
Where (P.Payment_Type='&v_Payment_Type' AND P.booking_id = B.booking_id) AND B.cust_id = C.cust_id;
TTitle off;
clear column;
