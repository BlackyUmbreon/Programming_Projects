cl scr
PROMPT 'Search Customers Details with Room Price'
PROMPT
PROMPT
ACCEPT v_Min_Room_Price  number  FORMAT '99999.99'   PROMPT 'Enter Minimum Room Price                        :'
ACCEPT v_Max_Room_Price  number  FORMAT '99999.99'   PROMPT 'Enter Maximum Room Price                      :'
PROMPT
PROMPT

Set linesize 180
Set pagesize 30 
Column Cust_ID format A10
Column Cust_FName format A20
Column Cust_LName format A20
Column Cust_Phone format A15
Column Cust_Email format A30
Column Booking_ID format A15
Column Room_Price format 99999.99
TTITLE Center 'Search Customer Detail with the Room Price within the Price Range input'
Select C.Cust_ID, C.Cust_FName, C.Cust_LName, Cust_Phone, Cust_Email, B.Booking_ID, R.Room_Price 
From room R, booking B, customer C, booking_detail BD
Where (((R.Room_Price BETWEEN &v_Min_Room_Price AND &v_Max_Room_Price) AND BD.Room_ID=R.Room_ID)AND BD.booking_id = B.booking_id) AND B.cust_id = C.cust_id
Order by R.Room_Price;
TTitle off;
clear column;
